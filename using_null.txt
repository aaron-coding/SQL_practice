#2
SELECT 
  teacher.name, dept.name
FROM 
  teacher 
INNER JOIN 
  dept
ON 
  (teacher.dept=dept.id)

  teacher.name, dept.name

  teacher

  dept
ON 
  (teacher.dept=dept.id)

  teacher.name, dept.name

  teacher

  (teacher.dept=dept.id)

#6
SELECT
  teacher.name, COALESCE(dept.name, 'None')
FROM
  teacher
LEFT OUTER JOIN
  dept

  dept.id = teacher.dept

#8
SELECT
 dept.name, COUNT(teacher.name)
FROM
  teacher
RIGHT JOIN
  dept
ON 
  teacher.dept = dept.id
GROUP BY
  dept.name

    ELSE 'Art'
  END
